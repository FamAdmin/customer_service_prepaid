def sizes=(new_sizes)
  prop_hash = properties.map{|prop| [prop['name'], prop['value']]}.to_h
  merged_hash = prop_hash.merge(new_sizes)
  self[:properties] = merged_hash.map{|k, v| {'name' => k, 'value' => v}}
end
